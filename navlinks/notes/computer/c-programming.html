2. WAP TO CHECK THE ENTERED NUMBER IS PALINDROME OR NOT.

#include <stdio.h> 
#include <conio.h>
void main(){
int n, m, Rem, Rev=0;
printf("Enter any number:");
scanf("%d", &n);
m=n;
while (n!=0){
    Rem=n%10;
    Rev=Rev*10+Rem;
    n=n/10;
}
if(Rev==m){
    printf("The entered number is palindrome");
}
else{
    printf("Entered number is not palindrome.");
}
getch();
}
