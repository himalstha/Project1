    printf("The greatest number is : %d", g);

    getch() ;

    return 0;

}



int greatest(int a, int b, int c)

{

    int g;

    if ( a >= b && a >= c)

    {

        g = a ;

    }

    else if ( b >= a && b>= c)

    {

        g = b ;

    }

    else

    {

        g = c ;

    }

    return g ;

}

int armstrong(int);

{

    int n, arm;

    printf ("Enter any number :");

    scanf ("%d", &n);

    arm = armstrong(n);

    if (arm == n)

    {

        printf("The number is armstrong.");

    }

    else

    {

        printf("The number is not armstrong.");

    }

}



int armstrong(int n)

{

    int arm =0, R;

    while (n>0)

    {

        R = n % 10;

        arm = arm + R*R*R;

        n=n/10;

    }

    return arm;

}
